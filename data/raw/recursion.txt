def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n - 1)


def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)


def sum_recursive(nums):
    if not nums:
        return 0
    return nums[0] + sum_recursive(nums[1:])


def power(base, exp):
    if exp == 0:
        return 1
    return base * power(base, exp - 1)


def count_down(n):
    if n < 0:
        return
    count_down(n - 1)
