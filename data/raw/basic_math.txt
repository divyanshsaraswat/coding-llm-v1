def add(a, b):
    return a + b


def subtract(a, b):
    return a - b


def multiply(a, b):
    return a * b


def divide(a, b):
    if b == 0:
        return None
    return a / b


def absolute(n):
    if n < 0:
        return -n
    return n


def is_even(n):
    return n % 2 == 0


def is_odd(n):
    return n % 2 != 0


def max_of_three(a, b, c):
    if a >= b and a >= c:
        return a
    if b >= a and b >= c:
        return b
    return c


def min_of_three(a, b, c):
    if a <= b and a <= c:
        return a
    if b <= a and b <= c:
        return b
    return c
